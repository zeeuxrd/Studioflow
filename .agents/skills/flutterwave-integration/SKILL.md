# Flutterwave Integration Skill

## Purpose
Enable StudioFlow to accept payments for digital products (eBooks, checklists, mini-courses, template packs) via Flutterwave – a leading payment gateway in Africa and emerging markets.

## When to Use
- User wants to sell a digital product generated by ProductAlchemist
- User sets a price ($1–$500, recommended $5–$20)
- Payment collection for one-time purchases (no subscriptions initially)

## Implementation Steps

### 1. Environment Variables

Ensure the following environment variables are set in your `.env` file:
```env
FLUTTERWAVE_PUBLIC_KEY="your_public_key_here"
FLUTTERWAVE_SECRET_KEY="your_secret_key_here"
FLUTTERWAVE_WEBHOOK_SECRET="your_webhook_secret_here"
```

### 2. Create Payment Link (Server-side)

Endpoint: `POST /api/payments/initialize`
Request body:
```json
{
  "product_id": "uuid",
  "product_title": "string",
  "price": 9.99,
  "user_email": "buyer@example.com",
  "user_id": "uuid"
}
```

import crypto from 'crypto';

// Types
interface FlutterwaveWebhookPayload {
  event: string;
  data: {
    id: string;           // transaction id
    tx_ref: string;       // your unique reference (store product_id + user_id)
    amount: number;
    currency: string;
    status: 'successful' | 'failed';
    customer: { email: string };
    meta?: { product_id: string; user_id: string };
  };
}

// Verify signature
function verifyWebhookSignature(
  payload: string,
  signatureHeader: string,
  secret: string
): boolean {
  const expectedSignature = crypto
    .createHmac('sha512', secret)
    .update(payload)
    .digest('hex');
  return crypto.timingSafeEqual(
    Buffer.from(signatureHeader),
    Buffer.from(expectedSignature)
  );
}

// Main handler (for Vercel / Next.js API route style)
export async function POST(req: Request) {
  const signature = req.headers.get('verif-hash');
  if (!signature) {
    return new Response('Missing signature', { status: 401 });
  }

  const rawBody = await req.text();
  const secret = process.env.FLUTTERWAVE_WEBHOOK_SECRET!;
  if (!verifyWebhookSignature(rawBody, signature, secret)) {
    return new Response('Invalid signature', { status: 401 });
  }

  const payload: FlutterwaveWebhookPayload = JSON.parse(rawBody);
  
  // Only process successful charge events
  if (payload.event !== 'charge.completed') {
    return new Response('Ignored event', { status: 200 });
  }

  const { data } = payload;
  if (data.status !== 'successful') {
    // Log failed transaction but don't grant access
